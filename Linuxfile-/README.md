# Why Linux?
Linux powers the cloud, servers, DevOps pipelines, containers, and most modern infrastructure. Learning the terminal is a superpower for developers and engineers.

# File & Directory Commands
pwd         # Show current directory

ls -la      # List files (detailed)

cd dir      # Change directory

mkdir app   # Create directory

rm -rf app  # Remove directory

cp a b       # Copy file

mv a b       # Move/Rename file

# Search & Text Processing
grep "text" file.txt      # Search text in file

find . -name "*.js"       # Find files

cat file.txt              # Show file content

nano file.txt             # Edit file

# System Monitoring
top        # Running processes

htop       # Better process viewer

df -h      # Disk usage

free -h    # RAM usage

uptime     # System uptime

# Networking Basics
ping google.com
curl https://example.com
ssh user@server-ip

# Package Management (Ubuntu/Debian)
sudo apt update
sudo apt upgrade
sudo apt install git

# Pro Tips
Use Tab for auto-completion.
Use Ctrl + R to search command history.
Chain commands with && to automate workflows.
Use man <command> to read documentation.
